                    [value1, value2, value3, value4, ...., valueN];

                    Array.of(value1, value2, value3);

    push() - To insert a new element into the array at last index position.
    pop() - To delete the element from last index position.
    toString() - It converts array into string format.
    splice(index, deleteCount, insertElement1, insertElement2, insertElement3, .... insertElementN);
    concat(secondArray); - Joins more than one array information.
    sort() - Sorts the data in ascending order.
    reverse() - It reverses the whole array elements.
    indexOf(value) - Finds index of an value.
    includes(value) - Checks whether value present in the array or not.
